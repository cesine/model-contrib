    Find the nodes within the specified rectangle.

        function search(x0, y0, x3, y3) {
      var selectedData = [];
      quadtree.visit(function(node, x1, y1, x2, y2) {
        var d = node.point, x, y;
        if (d) {
          x = node.x;
          y = node.y;
          d.visited = true;
          if(x >= x0 && x < x3 && y >= y0 && y < y3){
            d.selected = true;
            selectedData.push(d);
          }
        }
        return x1 >= x3 || y1 >= y3 || x2 < x0 || y2 < y0;
      });
      return selectedData;
    }
  };
